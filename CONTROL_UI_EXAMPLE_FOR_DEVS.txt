SPICE CREAM - пример контрольного UI для ручной настройки элементов

Идея:
Контрольный UI справа - это обычный HTML-блок поверх Phaser canvas. Он нужен только для тестовых локальных html, чтобы руками двигать элементы, менять scale/alpha/angle и потом скопировать готовый JSON-конфиг.

Обычно я делаю так:
1. В коде есть объект CONFIG с начальными координатами.
2. Phaser создает игровые объекты по CONFIG.
3. Справа создается HTML-панель с input type="range" / number / button.
4. При изменении input обновляется CONFIG и сразу вызывается applyConfig().
5. Кнопка Copy JSON копирует актуальный CONFIG в буфер.

Минимальный пример:

JS / Phaser:

const CONFIG = {
  panel: {
    x: 540,
    y: -8,
    scale: 1,
    angle: 0
  }
};

let panelImage;

class TestScene extends Phaser.Scene {
  preload() {
    this.load.image("panelBase", "./assets/PANEL_BASE.png");
  }

  create() {
    panelImage = this.add.image(0, 0, "panelBase").setOrigin(0.5, 0);
    applyConfig();
    createControlUI();
  }
}

function applyConfig() {
  const p = CONFIG.panel;
  if (!panelImage) return;

  panelImage
    .setPosition(p.x, p.y)
    .setScale(p.scale)
    .setAngle(p.angle);
}

function createControlUI() {
  const root = document.getElementById("control-ui");
  root.innerHTML = "";

  root.appendChild(makeTitle("Panel"));
  root.appendChild(makeSlider("x", CONFIG.panel, "x", 0, 1080, 1));
  root.appendChild(makeSlider("y", CONFIG.panel, "y", -300, 1920, 1));
  root.appendChild(makeSlider("scale", CONFIG.panel, "scale", 0.1, 5, 0.01));
  root.appendChild(makeSlider("angle", CONFIG.panel, "angle", -180, 180, 1));

  const copy = document.createElement("button");
  copy.textContent = "Copy JSON";
  copy.onclick = async () => {
    const text = JSON.stringify(CONFIG, null, 2);
    await navigator.clipboard.writeText(text);
    copy.textContent = "Copied!";
    setTimeout(() => copy.textContent = "Copy JSON", 700);
  };
  root.appendChild(copy);
}

function makeTitle(text) {
  const h = document.createElement("div");
  h.textContent = text;
  h.style.margin = "12px 0 6px";
  h.style.fontWeight = "700";
  return h;
}

function makeSlider(labelText, target, key, min, max, step) {
  const label = document.createElement("label");

  const name = document.createElement("span");
  name.textContent = labelText;

  const input = document.createElement("input");
  input.type = "range";
  input.min = min;
  input.max = max;
  input.step = step;
  input.value = target[key];

  const value = document.createElement("span");
  value.textContent = target[key];

  input.oninput = () => {
    target[key] = Number(input.value);
    value.textContent = target[key];
    applyConfig();
  };

  label.appendChild(name);
  label.appendChild(input);
  label.appendChild(value);
  return label;
}

Пример для массива элементов:

const CONFIG = {
  powerPoints: [
    { id: 1, x: 126, y: 117, scale: 1 },
    { id: 2, x: 160, y: 117, scale: 1 }
  ]
};

let powerPointImages = [];

function applyConfig() {
  CONFIG.powerPoints.forEach((cfg, index) => {
    const img = powerPointImages[index];
    if (!img) return;
    img.setPosition(cfg.x, cfg.y).setScale(cfg.scale);
  });
}

function createControlUI() {
  const root = document.getElementById("control-ui");
  root.innerHTML = "";

  CONFIG.powerPoints.forEach((point) => {
    root.appendChild(makeTitle("Power " + point.id));
    root.appendChild(makeSlider("x", point, "x", 0, 1080, 1));
    root.appendChild(makeSlider("y", point, "y", 0, 1920, 1));
    root.appendChild(makeSlider("scale", point, "scale", 0.1, 3, 0.01));
  });
}

Полезные правила:
- Все настройки хранить в одном CONFIG, чтобы его было легко копировать.
- applyConfig() должна быть единственным местом, где настройки реально применяются к Phaser-объектам.
- Для финального билда контрольный UI удаляется, а CONFIG остается в коде как готовые координаты.
- Для pixel art ассетов лучше использовать Phaser.CANVAS или pixelArt: true / roundPixels: true, если это нужно проекту.
- Если html открывается через file:// и ассеты грузятся в Phaser canvas, возможны зеленые квадраты или SecurityError. Для тестов надежнее запускать локальный сервер.
